void push(char ch)
    {
        if (TOP >= 100 - 1)
        {
            printf("Stack Overflow");
        }
        else
        {
            TOP++;
            stack[TOP] = ch;
        }
    }
char pop()
    {

        if (TOP <= -1)
        {
            printf("Stack Underflow");
        }
        else
        {
            return stack[TOP--];
        }
    }
