# p. 210 예제
```
#include <stdio.h>
#include <stdlib.h>
#define SNUM 10
#define CNUM 3
int main(void)
{
	int student[SNUM][CNUM];
	int i, j, sum;

	for (i = 0; i < SNUM; i++)
	{
		printf("자격증 입력: ");
		scanf_s("%d %d %d", &student[i][0], &student[i][1], &student[i][2]);
	}
	for (i = 0; i < SNUM; i++)
	{
		sum = 0;
		for (j = 0; j < CNUM; j++)
			sum += student[i][j];
		printf("%d번 학생 자격증 : %d\n", i, sum);
	}
	return 0;
}

# p. 212 예제

```
#include <stdio.h>
#include <stdlib.h>
#define SIZE1 10
#define SIZE2 5
int main(void)
{
	int product[SIZE1][SIZE2] = { 0 }; // 배열의 선언, 모두 0으로 초기화
	int i, j, sale;

	for(i=0; i<SIZE1;i++)
		for (j = 0; j < 3; j++)
		{
			sale = rand() % 5;
			product[i][sale]++;
		}
	for (i = 0; i < SIZE1; i++)
	{
		for (j = 0; j < SIZE2; j++)
			printf("%d번 제품: 하양 %d, 검정 %d, 노랑 %d, 빨강 %d, 파랑 %d", i, product[i][0], product[i][1], product[i][2], product[i][3], product[i][4]);
		printf("\n");
	}
	return 0;
}
```

# p.214 에제

```
#include <stdio.h>
#include <stdlib.h>
#define SIZE1 10
#define SIZE2 5
int main(void)
{
	int arr1[SIZE1][SIZE2];
	int arr2[SIZE1][SIZE2];
	int arr3[SIZE1][SIZE2];
	int i, j, sum;

	for(i=0; i<SIZE1; i++)
		for (j = 0; j < SIZE2; j++)
		{
			arr1[i][j] = rand() % 6;
			arr2[i][j] = rand() % 6;
			arr3[i][j] = 0;
		}

	for (i = 0; i < SIZE1; i++)
		for (j = 0; j < SIZE2; j++)
			arr3[i][j] = arr1[i][j] + arr2[i][j];

	printf("배열 1의 내용\n");
	for (i = 0; i < SIZE1; i++) {
		for (j = 0; j < SIZE2; j++)
			printf("%2d", arr1[i][j]);
		printf("\n");
	}

	printf("\n배열2의 내용\n");
	for (i = 0; i < SIZE2; i++) {
		for (j = 0; j < SIZE2; j++)
			printf("%2d", arr2[i][j]);
		printf("\n");
	}

	printf("\n배열3의 내용\n");
	for (i = 0; i < SIZE2; i++) {
		for (j = 0; j < SIZE2; j++)
			printf("%2d", arr3[i][j]);
		printf("\n");
	}

	return 0;
}
